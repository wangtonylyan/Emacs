1. bashrc.sh将作为所有shell的默认配置文件，须由各shell主动加载
 #+begin_src sh
# echo >> ~/.bashrc
if [ -f "$HOME/.emacs.d/my.config/system/bashrc.sh" ]; then
   source "$HOME/.emacs.d/my.config/system/bashrc.sh"
fi
 #+end_src
